00115 void FAT_player_playNoteWithTsp(note* note, u8 channel, u8 transpose) {
00116     if (note->freq != NULL_VALUE) {
00117         instrument* inst = &(FAT_tracker.allInstruments[note->instrument]);
00118         u16 sweepshifts = (inst->sweep & 0x70) >> 4;
00119         u16 sweeptime = (inst->sweep & 0x0F);
00120         u16 sweepdir = 1;
00121         if (sweeptime > 7) {
00122             sweeptime -= 8;
00123             sweepdir = 0;
00124         }
00125 
00126         switch (channel) {
00127             case 0: // PU1
00128                 //ham_DrawText (23, 16, "PU1");
00129                 snd_playSoundOnChannel1(
00130                         sweeptime, sweepdir, sweepshifts,
00131                         inst->envelope & 0x0f, (inst->envelope & 0x10)>>4, (inst->envelope & 0xe0) >> 5, inst->wavedutyOrPolynomialStep,
00132                         inst->soundlength, inst->loopmode,
00133                         note->freq, transpose + FAT_tracker.transpose);
00134                 break;
00135             case 1: // PU2
00136                 //ham_DrawText (23, 16, "PU2");
00137                 snd_playSoundOnChannel2(inst->envelope & 0x0f, (inst->envelope & 0x10)>>4, (inst->envelope & 0xe0) >> 5,
00138                         inst->wavedutyOrPolynomialStep,
00139                         inst->soundlength, inst->loopmode,
00140                         note->freq, transpose + FAT_tracker.transpose);
00141                 break;
00142 
00143             case 2: // WAV
00144                 snd_playSoundOnChannel3(inst->volumeRatio, inst->soundlength, inst->loopmode, inst->voiceAndBank & 0x1f,
00145                         (inst->voiceAndBank & 0x20) >> 5, (inst->voiceAndBank & 0x40) >> 6, note->freq, transpose + FAT_tracker.transpose);
00146                 break;
00147             case 3: // NOISE
00148                 //ham_DrawText (23, 16, "NOI");
00149                 snd_playSoundOnChannel4(inst->envelope & 0x0f, (inst->envelope & 0x10)>>4, (inst->envelope & 0xe0) >> 5, inst->soundlength,
00150                         inst->loopmode, note->note & 0x0f, inst->wavedutyOrPolynomialStep, note->freq / NB_FREQUENCES, transpose + FAT_tracker.transpose);
00151                 break;
00152         }
00153 
00154     }

00244 void FAT_player_timerFunc_playSequences() {
00245     tempoReach--;
00246     if (tempoReach <= 0) {
00247 
00248         for (u8 i = 0; i < 6; i++) {
00249             FAT_currentPlayedSequence = FAT_tracker.channels[i].sequences[actualSequencesForChannel[i]];
00250             if (FAT_currentPlayedSequence != NULL_VALUE) {
00251                 // lire la séquence actuelle
00252                 sequence* seq = &FAT_tracker.allSequences[FAT_currentPlayedSequence];
00253 
00254                 FAT_currentPlayedBlock = seq->blocks[actualBlocksForChannel[i]];
00255                 if (FAT_currentPlayedBlock != NULL_VALUE) {
00256                     block* block = &FAT_tracker.allBlocks[FAT_currentPlayedBlock];
00257 
00258                     // Déplacement des curseurs de lecture
00259                     FAT_player_moveOrHideCursor(i);
00260 
00261                     // TODO BUFFERISER
00262                     FAT_player_playNoteWithTsp
00263                             (&(block->notes[actualNotesForChannel[i]]), i, seq->transpose[actualBlocksForChannel[i]]);
00264 
00265                     actualNotesForChannel[i]++;
00266                     if (actualNotesForChannel[i] >= NB_NOTES_IN_ONE_BLOCK) {
00267                         actualNotesForChannel[i] = 0;
00268                         actualBlocksForChannel[i]++;
00269                         if (actualBlocksForChannel[i] >= NB_BLOCKS_IN_SEQUENCE
00270                                 || seq->blocks[actualBlocksForChannel[i]] == NULL_VALUE) {
00271                             actualBlocksForChannel[i] = 0;
00272                             actualSequencesForChannel[i]++;
00273                             if (actualSequencesForChannel[i] > NB_MAX_SEQUENCES
00274                                     || FAT_tracker.channels[i].sequences[actualSequencesForChannel[i]] == NULL_VALUE
00275                                     || FAT_data_isSequenceEmpty(FAT_tracker.channels[i].sequences[actualSequencesForChannel[i]])) {
00276                                 actualSequencesForChannel[i] = 0;
00277                                 // TODO ne pas remonter à zéro -> remonter à la premiere séquence dispo
00278                                 // si pas de séquences dispo -> NULL_VALUE
00279                             }
00280                         }
00281                     }
00282                 } else {
00283                     actualBlocksForChannel[i] = 0;
00284                     actualSequencesForChannel[i]++;
00285                     if (actualSequencesForChannel[i] > NB_MAX_SEQUENCES) {
00286                         actualSequencesForChannel[i] = 0;
00287                     }
00288                 }
00289 
00290             } else {
00291                 actualSequencesForChannel[i] = 0;
00292             }
00293 
00294         }
00295 
00296         tempoReach = ((60000 / FAT_tracker.tempo) / 4) - TEMPO_TIMER_HARDWARE_VALUE;

00411 
00412     switch (FAT_currentScreen) {

00414             FAT_player_hideBlockCursor();
00415             FAT_player_hideNoteCursor();

00417                 // la lecture a été lancée depuis l'écran SONG
00418                 // on dispose du numéro de ligne actuellement jouée dans actualSequencesForChannel[channel]
00419                 ham_SetObjXY(FAT_cursor_playerSequences_obj[channel],
00420                         23 + (channel * (8 + 16)), 
00421                         15 + ((actualSequencesForChannel[channel] - FAT_screenSong_currentStartLine)*8));
00422                 ham_SetObjVisible(FAT_cursor_playerSequences_obj[channel], 1);
00423             } else {
00424                 FAT_player_hideSequencesCursors();
00425             }
00426             break;
00427         case SCREEN_BLOCKS_ID: // on est dans l'écran BLOCKS
00428             FAT_player_hideSequencesCursors();
00429             FAT_player_hideNoteCursor();

00431                     FAT_currentPlayedSequence == FAT_screenBlocks_currentSequenceId) {
00432                 ham_SetObjXY(FAT_cursor_player2_obj, 23, 15 + (actualBlocksForChannel[channel]*8));

00434             } else {
00435                 FAT_player_hideBlockCursor();
00436             }
00437

00439         case SCREEN_NOTES_ID: // on est dans l'écran NOTES
00440             FAT_player_hideSequencesCursors();
00441             FAT_player_hideBlockCursor();
